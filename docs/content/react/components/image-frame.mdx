## Usage

```tsx
import { ImageFrame } from "@grape_design_react/react";
```

```tsx
<ImageFrame
  ratio={4 / 3}
  borderRadius="r2"
  stroke
  src="..."
  alt="..."
  fallback={<Skeleton />}
/>
```
